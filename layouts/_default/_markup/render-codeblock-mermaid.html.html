// Initialize Mermaid diagrams that are initially visible document.querySelectorAll(".mermaid:not([hidden])").forEach((diagram) => { mermaid.init(undefined, diagram); }); // MutationObserver to reinitialize diagrams on visibility change new MutationObserver((mutations) => { mutations.forEach((mutation) => { if ( mutation.type === "attributes" && mutation.attributeName === "hidden" && !mutation.target.hidden ) { mutation.target.querySelectorAll(".mermaid").forEach((diagram) => { mermaid.init(undefined, diagram); }); } }); }).observe(document.body, { attributes: true, subtree: true }); });
